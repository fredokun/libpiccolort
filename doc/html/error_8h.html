Detailed Description

Provides several macros and functions to handle errors.

Senseless example of use: int main() { ALLOC_ERROR(foo_err); foo(1, &foo_err); if (HAS_ERROR(foo_err)) CRASH(&foo_err); // or PRINT_ERROR(&foo_err) return EXIT_SUCCESS; }

int foo(int a, PICC_Error *error) { int f = 0; if (a < 0) { NEW_ERROR(error, ERR_WRONG_ARG); } else { ALLOC_ERROR(bar_err); int b = bar(-a, &bar_err); if (HAS_ERROR(bar_err)) ADD_ERROR(error, bar_err, ERR_FOO_FAILED); else f = a + b; } return f; }

int bar(int b, PICC_Error *error) { if (b < 0) NEW_ERROR(error, ERR_WRONG_ARG); else b++; return b; }
